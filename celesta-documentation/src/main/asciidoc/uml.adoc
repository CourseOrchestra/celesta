  +getFoo(): Integer
  +setFoo(Integer foo)
  +getBar(): String
  +setBar(String bar)
}

class FooCursor.Columns {
  +foo(): ColumnMeta<Integer>

FooCursor "COLUMNS" --> FooCursor.Columns
